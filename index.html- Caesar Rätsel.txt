    <script>
        document.getElementById("prüf-btn").addEventListener("click", function() {
            let richtigeAntwort = ["F", "U", "N", "K", "E", "L", "N"];
            let eingaben = document.getElementsByClassName("eingabefeld");
            let benutzerEingabe = Array.from(eingaben).map(input => input.value.toUpperCase());

            if (JSON.stringify(benutzerEingabe) === JSON.stringify(richtigeAntwort)) {
                document.getElementById("rückmeldung").textContent = "✅ Richtig! Das nächste Rätsel ist freigeschaltet!";
            } else {
                document.getElementById("rückmeldung").textContent = "❌ Falsche Antwort. Versuche es erneut.";
            }
        });
    </script>
